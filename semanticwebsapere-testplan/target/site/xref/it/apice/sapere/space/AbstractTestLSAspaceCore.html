253 	 * <p>
254 	 * Tests a normal scenario.
255 	 * </p>
256 	 * 
257 	 * @throws Exception
258 	 *             URI problem
259 	 */
260 	@Test
261 	public final void testScenario() throws Exception {
262 		try {
263 			final LSA lsa = createFactory().createLSA();
264 			final LSAObserver scObs = createScenarioObserver(lsa);
265 
266 			assertTrue(sObs.noEventOccurred());
267 
268 			// 1. Register observer
269 			space.observe(lsa.getLSAId(), scObs);
270 
271 			assertTrue(sObs.checkFirstOcc(SpaceOperationType.AGENT_ACTION));
272 
273 			// 2. Inject and read the same LSA
274 			final LSA rLsa = space.inject(lsa).read(lsa.getLSAId());
275 
276 			// (added to ignore additions from the reasoner)
277 			rLsa.getSemanticDescription().removeProperty(
278 					createFactory().createProperty(
279 							new URI("http://www.w3.org/2002/07/owl#sameAs")));
280 			assertEquals(lsa, rLsa);
281 
282 			assertTrue(sObs.checkFirstOcc(SpaceOperationType.AGENT_INJECT));
283 			assertTrue(sObs.checkFirstOcc(SpaceOperationType.AGENT_READ));
284 
285 			// 3. Update the LSA
286 			lsa.getSemanticDescription().addProperty(
287 					createFactory().createProperty(
288 							new URI("http://localhost:8080/sapere#prop"),
289 							createFactory().createPropertyValue(Boolean.TRUE)));
290 			space.update(lsa);
291 
292 			assertTrue(sObs.checkFirstOcc(SpaceOperationType.AGENT_UPDATE));
293 
294 			// 4. Remove the LSA
295 			space.remove(lsa);
296 
297 			assertTrue(sObs.checkFirstOcc(SpaceOperationType.AGENT_REMOVE));
298 
299 			// 5. Ignore the LSA
300 			space.ignore(lsa.getLSAId(), scObs);
301 
302 			assertTrue(sObs.checkFirstOcc(SpaceOperationType.AGENT_ACTION));
303 			assertTrue(sObs.noEventOccurred());
304 		} catch (SAPEREException e) {
305 			e.printStackTrace();
306 			fail("Something went wrong during normal scenario "
307 					+ "(see stdout for details)");
308 		}
309 	}

312 	 * <p>
313 	 * Creates an observer for the normal scenario.
314 	 * </p>
315 	 * 
316 	 * @param observedLSA
317 	 *            The LSA to be observed
318 	 * @return An LSAObserver
319 	 */
320 	protected final LSAObserver createScenarioObserver(final LSA observedLSA) {
321 		return new LSAObserver() {
322 
323 			/** Counts occurred events. */
324 			private transient int evCounter = 0;
325 
326 			/** First INJECT. */
327 			private static final transient int INJECT = 1;
328 
329 			/** First UPDATE. */
330 			private static final transient int UPDATE = 2;
331 
332 			/** Finally REMOVE. */
333 			private static final transient int REMOVE = 3;
334 
335 			@Override
336 			public void eventOccurred(final LSAEvent ev) {
337 				evCounter++;
338 
339 				assertEquals(ev.getLSA(), observedLSA);
340 
341 				switch (evCounter) {
342 				case INJECT:
343 					assertEquals(ev.getOperationType(),
344 							SpaceOperationType.AGENT_INJECT);
345 					break;
346 				case UPDATE:
347 					assertEquals(ev.getOperationType(),
348 							SpaceOperationType.AGENT_UPDATE);
349 					break;
350 				case REMOVE:
351 					assertEquals(ev.getOperationType(),
352 							SpaceOperationType.AGENT_REMOVE);
353 					break;
354 				default:
355 					fail("Unexpected LSAEvent: " + ev);
356 					break;
357 				}
358 			}
359 		};
360 	}

408 	private static class SpaceObserverTester implements SpaceObserver {
409 
410 		/** Event queue. */
411 		private final transient List<SpaceOperationType> queue = 
412 				new LinkedList<SpaceOperationType>();
413 
414 		@Override
415 		public void eventOccurred(final SpaceEvent ev) {
416 			queue.add(ev.getOperationType());
417 		}
418 
419 		/**
420 		 * <p>
421 		 * Checks if an event occurred.
422 		 * </p>
423 		 * 
424 		 * @param type
425 		 *            Type of interesting event
426 		 * @return True if occurred, false otherwise
427 		 */
428 		public boolean checkFirstOcc(final SpaceOperationType type) {
429 			final int idx = queue.indexOf(type);
430 			if (idx >= 0) {
431 				queue.remove(idx);
432 				return true;
433 			}
434 
435 			return false;
436 		}
437 
438 		/**
439 		 * <p>
440 		 * Clears event queue.
441 		 * </p>
442 		 */
443 		public void reset() {
444 			queue.clear();
445 		}
446 
447 		/**
448 		 * <p>
449 		 * Checks if queue is empty.
450 		 * </p>
451 		 * 
452 		 * @return True if empty, false otherwise
453 		 */
454 		public boolean noEventOccurred() {
455 			return queue.isEmpty();
