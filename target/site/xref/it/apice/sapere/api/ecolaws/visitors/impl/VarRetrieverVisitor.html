14  /**
15   * <p>
16   * This class models a visitors capable of extracting all variables contained in
17   * an eco-law.
18   * </p>
19   * 
20   * @author Paolo Contessi
21   * 
22   */
23  public class VarRetrieverVisitor extends AbstractEcolawVisitor {
24  
25  	/** Eco-law's variables list. */
26  	private final transient List<VarTerm<?>> vars;
27  
28  	/**
29  	 * <p>
30  	 * Builds a new Eco-law Visitor whose aim is extracting all variable
31  	 * contained in it.
32  	 * </p>
33  	 * 
34  	 * @param anEcolaw
35  	 *            The eco-law to be inspected
36  	 */
37  	public VarRetrieverVisitor(final Ecolaw anEcolaw) {
38  		super(anEcolaw);
39  		vars = new LinkedList<VarTerm<?>>();
40  	}
41  	
42  	@Override
43  	protected final void doVisit(final Rate<?> rate) {
44  		checkAndAddIfVar(rate.getRateValue());
45  	}
46  
47  	@Override
48  	protected final void doVisit(final Term<?> term) {
49  		checkAndAddIfVar(term);
50  	}
51  
52  	@Override
53  	protected void doVisit(final Filter filter) {
54  		// Nothing to do...
55  	}
56  
57  	@Override
58  	protected void doVisit(final Product product) {
59  		// Nothing to do...
60  	}
61  
62  	@Override
63  	protected void doVisit(final Reactant reactant) {
64  		// Nothing to do...
65  	}
66  
67  	/**
68  	 * <p>
69  	 * Retrieves all variables contained in the specified eco-law.
70  	 * </p>
71  	 * 
72  	 * @return The list of all eco-law's variables
73  	 */
74  	public final List<VarTerm<?>> getAllVariables() {
75  		vars.clear();
76  		getEcolaw().accept(this);
77  
78  		return vars;
79  	}
80  
81  	/**
82  	 * <p>
83  	 * Checks if the provided object is a Variable of the ecolaw; in that case
84  	 * adds the variable to the list.
85  	 * </p>
86  	 * 
87  	 * @param obj
88  	 *            The candidate variable
89  	 */
90  	final void checkAndAddIfVar(final Object obj) {
91  		if (obj instanceof VarTerm<?>) {
92  			final VarTerm<?> var = (VarTerm<?>) obj;
93  			if (var.isVar()) {
94  				vars.add(var);
95  			}
96  		}
97  	}
98  
99  	/**
100 	 * <p>
101 	 * Inspects the provided law and extracts all the variables.
102 	 * </p>
103 	 * 
104 	 * @param law
105 	 *            The eco-law to be inspected
106 	 * @return The list of all eco-law's variables
107 	 */
108 	public static final List<VarTerm<?>> retrieveAllVariables(
109 			final Ecolaw law) {
110 		return new VarRetrieverVisitor(law).getAllVariables();
111 	}
