254 	public final LSAspace observe(final LSAid lsaId, final LSAObserver obs) {
255 		if (lsaId == null) {
256 			throw new IllegalArgumentException("Invalid LSA-id");
257 		}
258 
259 		if (obs == null) {
260 			throw new IllegalArgumentException("Invalid LSAObserver");
261 		}
262 
263 		acquireWriteLock();
264 		try {
265 			if (!observers.containsKey(lsaId)) {
266 				observers.put(lsaId, new LinkedList<LSAObserver>());
267 			}
268 
269 			final List<LSAObserver> list = observers.get(lsaId);
270 			if (!list.contains(obs)) {
271 				list.add(obs);
272 			}
273 
274 			// Space observation
275 			notifySpaceOperation(String.format(
276 					"LSA observation STARTED: %s (observer=%s)", lsaId, obs),
277 					lsaId, SpaceOperationType.AGENT_ACTION);
278 
279 			return this;
280 		} finally {
281 			releaseLock();
282 		}

324 	@Override
325 	public final MatchResult[] match(final CompiledEcolaw law)
326 			throws SAPEREException {
327 		if (law == null) {
328 			throw new IllegalArgumentException(
329 					"Invalid compiled eco-law provided");
330 		}
331 
332 		final List<MatchResult> res = new LinkedList<MatchResult>();
333 
334 		acquireReadLock();
335 		try {
336 			// 1. Run query
337 			final ResultSet iter = execQuery(model, law.getMatchQuery());
338 
339 			while (iter.hasNext()) {
340 				final QuerySolution sol = (QuerySolution) iter.next();
341 				// 2. Extract bindings
342 				final MutableMatchResult match = new MutableMatchResultImpl(
343 						this);
344 				for (String varName : law.variablesNames()) {
345 					match.register(varName,
346 							extractValue(sol.getResource(varName)), 1.0);
347 				}
348 
349 				// 3. Return them for evaluation
350 				res.add(match);
351 			}
352 		} catch (Exception ex) {
353 			throw new SAPEREException("Cannot complete match process", ex);
354 		} finally {
355 			releaseLock();
356 		}
357 
358 		return res.toArray(new MatchResult[res.size()]);
359 	}

624 	 * </p>
625 	 * 
626 	 * @param cLsa
627 	 *            The compiled LSA to be reverted
628 	 * @return A SAPERE model LSA
629 	 * @throws SAPEREException
630 	 *             Cannot retrieve LSA due to parsing issues
631 	 */
632 	private LSA retrieveLSA(final CompiledLSA<StmtIterator> cLsa)
633 			throws SAPEREException {
634 		final Model tmp = ModelFactory.createDefaultModel();
635 
636 		tmp.add(cLsa.getStatements());
637 		if (tmp.isEmpty()) {
638 			throw new SAPEREException("LSA is empty");
639 		}
640 
641 		try {
642 			final Set<LSA> res = converter.parseJenaModel(tmp);
643 			if (res.isEmpty() || res.size() > 1) {
644 				throw new SAPEREException("Invalid compiled LSA: "
645 						+ "should contains information about a single LSA");
646 			}
647 
648 			for (LSA lsa : res) {
649 				return lsa;
650 			}
651 		} catch (Exception e) {
652 			throw new SAPEREException("Cannot parse LSA", e);
653 		}
654 
655 		return null;
656 	}
