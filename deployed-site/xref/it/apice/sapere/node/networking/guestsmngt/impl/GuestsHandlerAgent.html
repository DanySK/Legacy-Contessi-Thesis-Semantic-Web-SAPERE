213 	@Override
214 	protected final void behaviour(final NodeServices services) {
215 		try {
216 			final GuestMessage message = CommunicationUtils
217 					.receiveMessage(socket);
218 
219 			if (message != null) {
220 				final String ipAddress = socket.getInetAddress()
221 						.getHostAddress();
222 				ipMap.put(message.getMacAddress(), ipAddress);
223 
224 				switch (message.getType()) {
225 				case INJECT:
226 					handleInject(message, services);
227 					break;
228 				case RE_INJECT:
229 					handleReInject(message, services);
230 					break;
231 				case MOVE_AWAY:
232 					handleMove(message, services);
233 					break;
234 				case UPDATE:
235 					handleUpdate(message, services);
236 					break;
237 				case REFRESH:
238 					handleRefresh(message, services);
239 					break;
240 				case READ:
241 					handleRead(message, services);
242 					break;
243 				case OBSERVE_ONCE:
244 					handleOnceSubscription(message);
245 					break;
246 				case OBSERVE:
247 					handleSubscription(message);
248 					break;
249 				case CANCEL_SUBSCRIPTION:
250 					handleCancelSubscription(message);
251 					break;
252 				default:
253 					spy("Error: invalid message type");
254 					if (!CommunicationUtils.sendString(
255 							"error: wrong message code", socket)) {
256 						spy("Cannot send error message");
257 					}
258 				}
259 			}
260 		} catch (Exception ex) {
261 			LoggerFactoryImpl.getInstance().getLogger(RequestHandler.class)
262 			.warn("Exception on RequestHandler ", ex);
263 		} finally {
264 			try {
265 				socket.close();
266 			} catch (IOException e) {
267 				LoggerFactoryImpl.getInstance().getLogger(RequestHandler.class)
268 						.warn("Cannot close the socket", e);
269 			}
270 		}
271 	}

462 	@SuppressWarnings({ "rawtypes", "unchecked" })
463 	private void handleReInject(final GuestMessage message,
464 			final NodeServices services) {
465 		spy("RE-INJECT ");
466 
467 		final CompiledLSA locLsa = createGuestInfoLSA(message, services);
468 		final LSAid idLoc = locLsa.getLSAid();
469 
470 		final LSAspaceCore space = services.getLSAspace();
471 
472 		space.beginWrite();
473 		try {
474 			// Removing what there was before (avoiding duplicate LSA-id)
475 			try {
476 				final CompiledLSA readLsa = space.read(locLsa
477 						.getLSAid());
478 				space.remove(readLsa);
479 			} catch (SAPEREException e) {
480 				spy("Re-injecting something that is not in the space");
481 			}
482 
483 			space.inject(locLsa);
484 
485 			final CompiledLSA recvLsa = services.getLSACompiler().parse(
486 					message.getLSA(), RDFFormat.RDF_XML);
487 			recvLsa.assertProperty(URI.create(POS_PROP_URI), idLoc.getId());
488 			space.inject(recvLsa);
489 
490 			final LSAid recvId = recvLsa.getLSAid();
491 			locMap.put(recvId.toString(), idLoc.toString());
492 		} catch (SAPEREException e) {
493 			error("Cannot inject the LSA again", e);
494 		} finally {
495 			space.done();
496 		}
497 
498 		if (!CommunicationUtils.sendString("ack", socket)) {
499 			spy("Error on ack sending");
500 		}
501 	}

622 		} catch (Exception ex) {
623 			LoggerFactoryImpl.getInstance().getLogger(SendMessageThread.class)
624 					.warn("Error while sending the message", ex);
625 		} finally {
626 			if (socket != null) {
627 				try {
628 					socket.close();
629 				} catch (IOException e) {
630 					LoggerFactoryImpl.getInstance()
631 							.getLogger(SendMessageThread.class)
632 							.error("Cannot close sending socket", e);
633 				}
634 			}
