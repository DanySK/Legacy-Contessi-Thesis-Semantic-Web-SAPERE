64  	@Override
65  	public final LSAspace inject(final LSA lsa) throws SAPEREException {
66  		if (lsa == null) {
67  			throw new IllegalArgumentException("Invalid LSA");
68  		}
69  
70  		if (read(lsa.getLSAId()) != null) {
71  			throw new SAPEREException("LSA already injected");
72  		}
73  
74  		converter.serializeLSA(lsa, model);
75  
76  		notifySpaceOperation(String.format("Injecting LSA: %s", lsa),
77  				lsa.getLSAId(), SpaceOperationType.AGENT_INJECT);
78  		notifyLSAObservers("LSA Injected", lsa, 
79  				SpaceOperationType.AGENT_INJECT);
80  
81  		return this;
82  	}

84  	@Override
85  	public final LSA read(final LSAid lsaId) throws SAPEREException {
86  		if (lsaId == null) {
87  			throw new IllegalArgumentException("Invalid LSA-id");
88  		}
89  
90  		final Resource rLsa = model.getResource(lsaId.getId().toString());
91  		if (rLsa != null) {
92  			try {
93  				final LSA lsa = converter.parseLSA(rLsa, model);
94  
95  				notifySpaceOperation(
96  						String.format("Reading LSA: id=%s", lsaId), lsaId,
97  						SpaceOperationType.AGENT_READ);
98  
99  				return lsa;
100 			} catch (Exception e) {
101 				throw new SAPEREException(String.format(
102 						"Cannot read LSA: id=%s", lsaId), e);
103 			}
104 		}
105 
106 		throw new SAPEREException(String.format("Unknown LSA: id=%s", lsaId));
107 	}

367 	 * @param lsa

375 	private LSAspace doUpdate(final LSA lsa, final SAPEREAgent requestor)
376 			throws SAPEREException {
377 		// Removes LSA from the space, then re-injects it..
378 		model.getResource(lsa.getLSAId().getId().toString()).removeProperties();
379 		converter.serializeLSA(lsa, model);
380 
381 		if (requestor != null) {
382 			notifySpaceOperation(String.format(
383 					"Updating LSA: id=%s, content=%s (requestor: %s)",
384 					lsa.getLSAId(), lsa.getSemanticDescription(), requestor),
385 					lsa.getLSAId(), SpaceOperationType.AGENT_UPDATE);
386 		} else {
387 			notifySpaceOperation(String.format(
388 					"Updating LSA: id=%s, content=%s (requestor: SYSTEM)",
389 					lsa.getLSAId(), lsa.getSemanticDescription()),
390 					lsa.getLSAId(), SpaceOperationType.SYSTEM_ACTION);
391 		}
392 		notifyLSAObservers("LSA Updated", lsa, SpaceOperationType.AGENT_UPDATE);
393

398 	 * <p>
399 	 * Executes OBSERVE operation.
400 	 * </p>
401 	 * 
402 	 * @param lsaId
403 	 *            The LSA-id of involved LSA
404 	 * @param requestor
405 	 *            The operation requestor
406 	 * @param obs
407 	 *            The observer involved
408 	 * @return Reference to this space
409 	 */
410 	private LSAspace doObserve(final LSAid lsaId, final SAPEREAgent requestor,
411 			final LSAObserver obs) {
412 		if (!observers.containsKey(lsaId)) {
413 			observers.put(lsaId, new LinkedList<LSAObserver>());
414 		}
415 
416 		observers.get(lsaId).add(obs);
417 
418 		if (requestor != null) {
419 			notifySpaceOperation(String.format(
420 					"Start observing LSA: id=%s (requestor: %s)", lsaId,
421 					requestor), lsaId, SpaceOperationType.AGENT_ACTION);
422 		} else {
423 			notifySpaceOperation(String.format("Start observing LSA: id=%s "
424 					+ "(requestor: SYSTEM)", lsaId), lsaId,
425 					SpaceOperationType.SYSTEM_ACTION);
426 		}
427 
428 		return this;
429 	}

433 	 * Executes IGNORE operation.
434 	 * </p>
435 	 * 
436 	 * @param lsaId
437 	 *            The LSA-id of involved LSA
438 	 * @param requestor
439 	 *            The operation requestor
440 	 * @param obs
441 	 *            The observer involved
442 	 * @return Reference to this space
443 	 */
444 	private LSAspace doIgnore(final LSAid lsaId, final SAPEREAgent requestor,
445 			final LSAObserver obs) {
446 		final List<LSAObserver> obss = observers.get(lsaId);
447 		if (obss != null) {
448 			// Removes it if is registered, otherwise nothing changes...
449 			obss.remove(obs);
450 
451 			if (requestor != null) {
452 				notifySpaceOperation(String.format(
453 						"Stop observing LSA: id=%s (requestor: %s)", lsaId,
454 						requestor), lsaId, SpaceOperationType.AGENT_ACTION);
455 			} else {
456 				notifySpaceOperation(String.format("Stop observing LSA: id=%s "
457 						+ "(requestor: SYSTEM)", lsaId), lsaId,
458 						SpaceOperationType.SYSTEM_ACTION);
459 			}
460 		}
461 
462 		return this;
463 	}
