137 			public void caretUpdate(CaretEvent e) {
138 				// Lets start with some default values for the line and column.
139 				int linenum = 1;
140 				int columnnum = 1;
141 
142 				// We create a try catch to catch any exceptions. We will simply
143 				// ignore such an error for our demonstration.
144 				try {
145 					// First we find the position of the caret. This is the
146 					// number of where the caret is in relation to the start of
147 					// the JTextArea
148 					// in the upper left corner. We use this position to find
149 					// offset values (eg what line we are on for the given
150 					// position as well as
151 					// what position that line starts on.
152 					int caretpos = queryInput.getCaretPosition();
153 					linenum = queryInput.getLineOfOffset(caretpos);
154 
155 					// We subtract the offset of where our line starts from the
156 					// overall caret position.
157 					// So lets say that we are on line 5 and that line starts at
158 					// caret position 100, if our caret position is currently
159 					// 106
160 					// we know that we must be on column 6 of line 5.
161 					columnnum = caretpos
162 							- queryInput.getLineStartOffset(linenum);
163 
164 					// We have to add one here because line numbers start at 0
165 					// for getLineOfOffset and we want it to start at 1 for
166 					// display.
167 					linenum += 1;
168 				} catch (Exception ex) {
169 					ex.printStackTrace();
170 				}
171 
172 				// Once we know the position of the line and the column, pass it
173 				// to a helper function for updating the status bar.
174 				updateStatus(linenum, columnnum);
175 			}

349 	private void printOKOutput(final ResultSet iter) {
350 		int counter = 1;
351 		while (iter != null && iter.hasNext()) {
352 			final QuerySolution sol = iter.nextSolution();
353 			final Iterator<?> vars = sol.varNames();
354 			if (vars != null && vars.hasNext()) {
355 				txtOutput.append("============== Variable Bindings (#"
356 						+ counter++ + ") ==============\n");
357 				while (vars.hasNext()) {
358 					final String var = (String) vars.next();
359 					txtOutput.append(String.format("?%s = %s\n", var,
360 							sol.get(var).toString()));
361 				}
362 			}
363 		}
364 
365 		printRDFGraphContent();
366 	}

374 		final StringWriter strW = new StringWriter();
375 		try {
376 			model.write(strW, "TURTLE");
377 			final String content = strW.toString();
378 			if (content.length() == 0) {
379 				txtOutput.append("(empty)\n");
380 			} else {
381 				txtOutput.append(content);
382 			}
383 		} catch (Exception ex) {
384 			txtOutput.append("ERROR: Cannot retrieve actual RDF Graph status");
385 		} finally {
386 			if (strW != null) {
387 				try {
388 					strW.close();
389 				} catch (Exception e) {
390 					e.printStackTrace();
391 				}
392 			}
393 		}
394 		txtOutput
395 				.append("===================================================\n\n");
396 	}
